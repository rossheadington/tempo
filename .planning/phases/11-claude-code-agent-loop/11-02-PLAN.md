```python
@dataclass(frozen=True, slots=True)
class AgentTurn:
    text: str               # the assistant's final reply, already concatenated
    session_id: str         # session id to persist (new or resumed)
    tokens_in: int          # input_tokens from the final ResultMessage usage block
    tokens_out: int         # output_tokens from the final ResultMessage usage block
    cost_usd: float | None  # total_cost_usd from ResultMessage, or None when SDK omits it
    duration_s: float       # wall-clock seconds for the whole run_turn call

class AgentInvocationError(Exception):
    """Raised when the SDK / claude CLI cannot be invoked (missing CLI, auth missing).
    Plan 11-03 catches this in voice_handler / text_handler and replies with a clear
    'Claude Code isn't running. Try `claude login` in a terminal.' message."""

async def run_turn(
    prompt: str,
    session_id: str | None,
    *,
    cwd: Path,
) -> AgentTurn: ...

def format_for_telegram(text: str) -> list[str]:
    """HTML-escape `text` (& < > only), then split into chunks <= 4096 chars.
    If splitting is needed, prefix each chunk with `[k/N] ` (k starts at 1).
    Split on `\\n\\n` boundaries when possible; hard-split as a last resort.
    Returns at least one chunk; an empty input becomes `[""]` (caller decides
    whether to send anything). The prefix counts toward the 4096 cap."""
```
