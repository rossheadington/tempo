From tempo/bot/sessions.py (new module — this plan creates it):
```python
SESSION_WINDOW_HOURS: int = 4

def get_or_create_session(
    conn: sqlite3.Connection,
    chat_id: int,
    *,
    now: datetime | None = None,
    window_hours: int = SESSION_WINDOW_HOURS,
) -> str | None:
    """Return the stored session_id if `last_message_at` is within window_hours of `now`,
    else None. `now` defaults to `datetime.now(UTC)` when omitted."""

def save_session(
    conn: sqlite3.Connection,
    chat_id: int,
    session_id: str,
    now: datetime | None = None,
) -> None:
    """UPSERT the (chat_id, session_id, last_message_at=now) row. `started_at` is set
    on the initial insert and preserved on subsequent UPDATEs for the same session_id;
    when `session_id` changes for the same chat_id, treat as a new session and reset
    `started_at` to `now`."""

def reset_session(conn: sqlite3.Connection, chat_id: int) -> None:
    """DELETE the row for chat_id; idempotent (no-op when absent)."""
```
